# Button

Buttons are created by adding the `.button` class to a `<button>`.

You can customize the size of buttons with the `.is-large`, `.is-medium` and `.is-small` classes.

export const buttonType = (size) => {
  let sizeClass = size ? ` is-${size}` : ''
  return `
<button class="button${sizeClass}">Button</button>
<button class="button${sizeClass}"><i class="icon external-link"></i>&emsp;Button</button>`
}
