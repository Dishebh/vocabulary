# Input

Inputs are created by adding the `.input` class to an `<input>`.

You can customize the size of inputs with the `.is-large` and `.is-medium` classes.

export const inputType = (size) => {
  let sizeClass = size ? ` is-${size}` : ''
  return `
<input class="input${sizeClass}" type="text" placeholder="Placeholder text"/>
<br/><br/>
<input class="input${sizeClass}" type="text" value="Typed text"/>
<br/><br/>
<input class="input${sizeClass}" type="text" value="Fixed text" disabled/>`
}
